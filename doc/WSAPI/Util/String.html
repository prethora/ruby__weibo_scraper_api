def self.gen_random_key(len = 16)
    table = ('a'..'z').to_a + ('0'..'9').to_a
    (0...len).map { table[rand(table.length)] }.join
end

def self.simple_parse(str,open,close)
    b = str.index(open)
    return nil if b.nil?
    b+= open.length
    e = str.index(close,b)
    return nil if e.nil?
    str[b..e-1]
end
